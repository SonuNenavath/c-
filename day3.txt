1.Addition of two 3*3 matrices
#include<stdio.h>
#include<math.h>
int main() {
    int a[3][3]={{1,0, 1},
                 {0,0,1}, 
                 {1,1,0}};
    int b[3][3]={{1,1,1}, 
                 {0,1,0}, 
                 {1,0,0}};
    int c[3][3],i,j;
    printf("\nThe first matrix is\n");
    for(i=0;i<3;i++){
        printf("\n");
        for(j=0;j<3;j++)
            printf("%d\t",a[i][j]);
    }
    printf("\nThe second matrix is\n");
    for(i=0;i<3;i++){
        printf("\n");
        for(j=0;j<3;j++)
            printf("%d\t",b[i][j]);
    }
    for(i=0;i<3;i++)
        for(j=0;j<3;j++)
            c[i][j]=a[i][j]+b[i][j];
    printf("\nThe addition of two matrices is\n");
    for(i=0;i<3;i++){
        printf("\n");
        for(j=0;j<3;j++)
            printf("%d\t",c[i][j]);
    }
    return 0;
}

3. Sorting 
#include<stdio.h>

int main()
{
    char a[5]={'z', 44, 66, 'a', 192};
    int i,j,temp,num;
    for(i=0;i<5;i++)
    {
        for(j=i+1;j<5;j++)
        {
            if(a[i]>a[j])
            {
                temp=a[i];
                a[i]=a[j];
                a[j]=temp;
            }
        }
    }
    printf("The sorted elements are: ");
    for(i=0;i<5;i++)
        printf("%c\t",a[i]);
    return 0;
}
